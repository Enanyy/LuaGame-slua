--测试继承和重写
A = Class("A")
function A:print()
	print("A:print")
end
function A:init(a)
	print("Class A a =" ..a)
end

B = Class("B",A)

function B:ctor()
	
end

function B:init(b)

	self.base:init(b)
	print("Class B b =" ..b)

end

C = Class("C",B)
function C:ctor()

end

function C:init(c)

	self.base:init(c)
	print("Class C c =" ..c)

end

function TestOverWrite()

	local a = A.new()
	a:print()

	--没有重写 调用A的print
	local b = B.new()
	b:init(111)

	local c = C.new()
	c:init(222)

end
