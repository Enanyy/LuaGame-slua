
require("Window")
require("UI_WidgetBase")

--UI_Widget继承于Window
UI_Widget = Class("UI_Widget",Window)



local this = UI_Widget

function this:ctor( path)
    self.path  = path
    self.windowType = WindowType.Widget --widget界面
    self.useMask = false --可以点击背景后面
    self.depth = 5000      --可以设置depth

end

function this:Start()

    UI_WidgetGen(self)

   

    LuaHelper.AddClick(self.mMainWindow,function()
    
        print("Click MainWindow")
        WindowManager:Open(UI_Main)

    end)

    

    
    local closeButton = self.mClose:GetComponent(UIButton) 

    if closeButton then
        closeButton.onClick:Add(EventDelegate(function() 
            print("Click Close")        
            self:Close()
        end))
    end
end
